•Les phases d'un projet(C Ip Ho T F G A):
    1-Cadrage (ou Analyse): comprendre le besoin du Client ET REMPLIR LES ANCIENNES DONNEES DU CLIENT  
    2-Instalation et parmetrage :
        -installation puis paramétrage de SageX3 cher le client .
    3-Homolgation: Validation du paramétrage par le client
    4-Test: Simulation de l'application avec les données du client
    5-Formation: Apprentissage de la solution au client ou a ses employés
    6-GO LIFE: debut d'utilisation de la solution par le client ou ses employés
    7-Assistance: accompagner le client durant une certaine periode de temps

NB: Toujours faire un livrable entre chaque phase et doit etre signer par le Client avant de passer
à la seconde phase.

            WITH BASE AS (
		SELECT 
			LEDTYP_0, CPY_0, FCY_0, FIY_0, ACC_0, CURLED_0, COA_0, ACCNUM_0,BPR_0,

			DEBLED_0, DEBLED_1, DEBLED_2, DEBLED_3, DEBLED_4, DEBLED_5,
			DEBLED_6, DEBLED_7, DEBLED_8, DEBLED_9, DEBLED_10, DEBLED_11, DEBLED_12,

			CDTLED_0, CDTLED_1, CDTLED_2, CDTLED_3, CDTLED_4, CDTLED_5,
			CDTLED_6, CDTLED_7, CDTLED_8, CDTLED_9, CDTLED_10, CDTLED_11, CDTLED_12

		FROM [X3Schema].BALANCE
		WHERE LEN(FCY_0) <> 0
		and (
			   (CAST(LEFT(ACC_0, 1) AS INT) = 4 and len(BPR_0) <> 0) OR 
			   (CAST(LEFT(ACC_0, 1) AS INT) <> 4 and len(BPR_0) = 0)
		)
	),

	DEB AS (
		SELECT 
			LEDTYP_0, CPY_0, FCY_0, FIY_0, ACC_0, CURLED_0, COA_0, ACCNUM_0,BPR_0,
			CAST(REPLACE(MOIS, 'DEBLED_', '') AS INT) AS MOIS,
			DEBMOIS
		FROM BASE
		UNPIVOT (
			DEBMOIS FOR MOIS IN (
				DEBLED_0, DEBLED_1, DEBLED_2, DEBLED_3, DEBLED_4, DEBLED_5,
				DEBLED_6, DEBLED_7, DEBLED_8, DEBLED_9, DEBLED_10, DEBLED_11, DEBLED_12
			)
		) AS U
	),

	CDT AS (
		SELECT 
			LEDTYP_0, CPY_0, FCY_0, FIY_0, ACC_0, CURLED_0, COA_0, ACCNUM_0,BPR_0,
			CAST(REPLACE(MOIS, 'CDTLED_', '') AS INT) AS MOIS,
			CDTMOIS
		FROM BASE
		UNPIVOT (
			CDTMOIS FOR MOIS IN (
				CDTLED_0, CDTLED_1, CDTLED_2, CDTLED_3, CDTLED_4, CDTLED_5,
				CDTLED_6, CDTLED_7, CDTLED_8, CDTLED_9, CDTLED_10, CDTLED_11, CDTLED_12
			)
		) AS U
	)

	SELECT 
		DEB.LEDTYP_0,
		DEB.CPY_0,
		DEB.FCY_0,
		DEB.FIY_0,
		DEB.ACC_0,
		DEB.CURLED_0,
		DEB.COA_0,
		DEB.ACCNUM_0,
		DEB.MOIS,
		DEB.BPR_0,
		DEB.DEBMOIS,
		CDT.CDTMOIS,
		DEB.DEBMOIS-CDT.CDTMOIS SOLDEMOIS
	FROM DEB
	INNER JOIN CDT 
		ON CDT.LEDTYP_0 = DEB.LEDTYP_0
		AND CDT.CPY_0    = DEB.CPY_0
		AND CDT.FCY_0    = DEB.FCY_0
		AND CDT.FIY_0    = DEB.FIY_0
		AND CDT.ACC_0    = DEB.ACC_0
		AND CDT.CURLED_0 = DEB.CURLED_0
		AND CDT.COA_0    = DEB.COA_0
		AND CDT.ACCNUM_0 = DEB.ACCNUM_0
		AND CDT.MOIS     = DEB.MOIS
		AND CDT.BPR_0     = DEB.BPR_0
'
